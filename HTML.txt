// 分享
function shareLink(){
  const text = "我都用這個看 基北高中職交通 👉";
  const url = window.location.href;

  if(navigator.share){
    navigator.share({ title:"基北高中職交通", text:text, url:url });
  }else{
    prompt("複製連結分享：", url);
  }
}

// 記錄學校
function saveSchool(){
  let school = prompt("輸入剛剛點的學校名稱");
  if(!school) return;

  let list = JSON.parse(localStorage.getItem("schools") || "[]");

  if(!list.includes(school)){
    list.push(school);
  }

  localStorage.setItem("schools", JSON.stringify(list));

  alert("已記錄：" + school);
}

// 比較
function compareSchools(){
  let list = JSON.parse(localStorage.getItem("schools") || "[]");

  let s1, s2;

  if(list.length >= 2){
    s1 = list[list.length-2];
    s2 = list[list.length-1];
  }else{
    s1 = prompt("第一間學校");
    if(!s1) return;

    s2 = prompt("第二間學校");
    if(!s2) return;
  }

  let origin = prompt("你的目的地（例如：建國中學）");
  if(!origin) return;

  openCompare(origin, s1, s2);
}

// 快速比較
function quickCompare(s1, s2){
  let origin = prompt("輸入住家（例如：台北市忠孝東路一段1號）");
  if(!origin) return;

  openCompare(origin, s1, s2);
}

// 開啟比較
function openCompare(origin, s1, s2){
  let url1 = "https://www.google.com/maps/dir/?api=1&origin=" 
    + encodeURIComponent(origin)
    + "&destination=" + encodeURIComponent(s1)
    + "&travelmode=transit";

  let url2 = "https://www.google.com/maps/dir/?api=1&origin=" 
    + encodeURIComponent(origin)
    + "&destination=" + encodeURIComponent(s2)
    + "&travelmode=transit";

  window.open(url1, "_blank");
  setTimeout(()=>window.open(url2, "_blank"), 500);
}
